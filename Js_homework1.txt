1. 'number' + 3 + 3 // number6   str конкатенуються з числами, які були додані між собою;

2. null + 3 // поверне 3   тому що null конвертується в 0;

3. 5 && "qwerty" // при порівнянні рядка з числом, нечисловий рядок конвертується в NaN, що завжди є false;

4. +'40' + +'2' + "hillel" // відбувається додавання значень string між собою "+42hillel";

5. '10' - 5 === 6 // '10' - 5 поверне 5, тому що '10' конвертується в 10    але 5 не строго дорівнює 6, тому значення false;

6. true + false // true приймаємо як 1, false приймаємо як 0, тоді весь вираз дорівнює 1;

7. '4px' - 3// NaN, таке віднімання є помилковим;

8. '4' - 3// поверне 1, бо '4' перетвориться на 4, а 4-3 поверне 1;

9. '6' + 3 ** 0// 3 ** 0 має пріоритет, після виконання дії буде 0  '6' str конкантинується з 0 num, що дорівнюватиме '60';

10. 12 / '6' // число ділимо на цифровий рядок, де '6' перетворюється на 6. 12/6 отримуємо 2;

11. '10' + (5 === 6) // дужки мають пріоритет, отримуємо false  потім str+false конкантинується і булеве значееня перетворюється в рядок '10false';

12. null == '' //null буде нульовим значенням, пустий рядок теж буде вважатися нульовим значенням, тож можемо сказати, що пуста строка завжди приведе до false;

13. 3 ** (9 / 3) // дужки мають пріоритет, отримуємо 3   Потім 3 підносимо в 3 степені отримаємо 27;

14. !!'false' == !!'true' // обидва значення будуть хибними, тому отримаємо (!false == !false) тепер також можна сказати що це true == true, отримаємо true;

15. 0 || '0' && 1 // && мають вищий пріоритет отримаємо 1, потім викорнуємо || і отримаємо 1;

16. (+null == false) < 1 // в дужках null ==false, вираз буде true. True приймаємо як 1, тоді 1<1 і це буде false;

17. false && true || true // && мають вищий пріоритет, false && true поверне true, потім true або true поверне true;

18. false && (false || true) // дужки мають пріоритет false або true поверне true,  false && true поверне нам false;

19. (+null == false) < 1 ** 5// в дужках отримаємо true, 1 в 5 ступені дорівнює 1, true<1, що дорівнює false;
