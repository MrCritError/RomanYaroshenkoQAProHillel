function findValueByKey(companyName) {
  function findCompany(clients) {
    for (const client of clients) {
      if (client.name === companyName) {
        return client;
      }
      if (client.partners) {
        const found = findCompany(client.partners);
        if (found) return found;
      }
    }
    return null;
  }
  
  return findCompany(company.clients);
}
